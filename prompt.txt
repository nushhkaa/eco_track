### ── ENGINE LOGIC INTERFACES ──

#### Create File: `js/calculator_engine.js`
export class CalculatorEngine {
  constructor() {
    this.factors = null;
  }

  async init() {
    const res = await fetch('./data/emission_factors.json');
    const data = await res.json();
    this.factors = data.factors;
  }

  calculateModule(category, subKey, value) {
    if (!this.factors || !this.factors[category]) return 0;
    const factor = this.factors[category][subKey] || 0;
    return value * factor; // outputs directly in kg CO2e
  }
}

#### Create File: `js/storage_manager.js`
export class StorageManager {
  static saveQuarterlyLog(grade, dataPayload) {
    const history = this.getHistory();
    const timestamp = new Date().toISOString().substring(0, 10);
    history.push({ timestamp, grade, data: dataPayload });
    localStorage.setItem('ecotracks_history', JSON.stringify(history));
  }

  static getHistory() {
    return JSON.parse(localStorage.getItem('ecotracks_history')) || [];
  }

  static getLatestForGrade(grade) {
    const history = this.getHistory();
    const filtered = history.filter(item => item.grade === grade);
    return filtered.length ? filtered[filtered.length - 1] : null;
  }

  static getPreviousForGrade(grade) {
    const history = this.getHistory();
    const filtered = history.filter(item => item.grade === grade);
    return filtered.length > 1 ? filtered[filtered.length - 2] : null;
  }

  static exportToCSV() {
    const history = this.getHistory();
    if (!history.length) return "No data available";
    
    let csvContent = "data:text/csv;charset=utf-8,Timestamp,Grade,Category,MetricValue,Computed_kgCO2e\n";
    history.forEach(row => {
      Object.keys(row.data).forEach(cat => {
        csvContent += `${row.timestamp},${row.grade},${cat},${row.data[cat].value},${row.data[cat].co2}\n`;
      });
    });
    return encodeURI(csvContent);
  }
}

      Object.keys(data).forEach(cat => {
        const currentKg = data[cat].co2;
        const pastKg = prevData ? prevData[cat].co2 : 0;

        const block = document.createElement('div');
        block.innerHTML = `
